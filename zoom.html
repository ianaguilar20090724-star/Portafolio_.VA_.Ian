import cv2

def ampliar():
    # Cargar la imagen
    io = cv2.imread("JEW.jfif")
    
    # Variable para el valor del zoom inicial
    zoom = 1.0
    
    while True:
        # Redimensionar con el factor de zoom actual
        ig = cv2.resize(io, None, fx=zoom, fy=zoom)
        
        # Mostrar la imagen
        cv2.imshow("Imagen", ig)
        
        # Esperar acci√≥n del usuario
        t = cv2.waitKey(0)
        
        if t == ord('+'):
            # Aumentar zoom 20%
            zoom = zoom * 1.2
            print(f"Zoom: {zoom:.2f}x")
            
        elif t == ord('-'):
            # Reducir zoom 0.2
            zoom = zoom - 0.2
            print(f"Zoom: {zoom:.2f}x")
            
        elif t == ord('o'):
            # Resetear a zoom original
            zoom = 1.0
            print("Zoom reseteado a 1.0x")
            
        elif t == ord('x'):
            # Salir del programa
            break
    
    cv2.destroyAllWindows()

# Ejecutar la funci√≥n
ampliar()

üìö Funciones Principales

Redimensiona una imagen usando dimensiones absolutas o factores de escala. En este caso usamos factores para zoom din√°mico.

Par√°metros clave:
‚Ä¢ imagen - Imagen a redimensionar
‚Ä¢ None - Cuando se usan factores en vez de tama√±o fijo
‚Ä¢ fx - Factor de escala horizontal

‚Ä¢ fx=1.0, fy=1.0 ‚Üí Tama√±o original (100%)
‚Ä¢ fx=2.0, fy=2.0 ‚Üí Doble tama√±o (200%)
‚Ä¢ fx=0.5, fy=0.5 ‚Üí Mitad del tama√±o (50%)
‚Ä¢ fx=1.5, fy=1.5 ‚Üí Una vez y media (150%)

import cv2

def zoom_con_mouse(event, x, y, flags, param):
    global zoom
    
    if event == cv2.EVENT_MOUSEWHEEL:
        if flags > 0:  # Scroll hacia arriba
            zoom = min(zoom * 1.1, 5.0)
        else:  # Scroll hacia abajo
            zoom = max(zoom / 1.1, 0.1)

io = cv2.imread("imagen.jpg")

    cv2.imshow("Zoom con Mouse", ig)
    
    if cv2.waitKey(1) & 0xFF == ord('x'):
